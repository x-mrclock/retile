district_embassy_arcology = {
	base_buildtime = @base_buildtime
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		is_planet_class = pc_city
	}
	
	potential = {
		is_planet_class = pc_city
	}
	
	allow = {
		retile_diplomacy = yes
		hidden_trigger = {
			retile_ready = yes
		}
	}

	resources = {
		category = planet_districts
		cost = {
			alloys = @base_cost
		}
		upkeep = {
			energy = @base_upkeep
			#alloys = 1
		}		
	}	

	triggered_planet_modifier = {		# Regular Jobs
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			planet_housing_add = 2
			job_diplomat_add = 2
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_ascension_perk = ap_arcology_project
				is_regular_empire = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			planet_housing_add = 1
			job_diplomat_add = 1
		}
	}
	triggered_planet_modifier = {		# Gestalt Jobs
		potential = {
			exists = owner
			owner = {
				is_gestalt = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			planet_housing_add = 2
			#job_diplomat_drone_add = 2
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_ascension_perk = ap_arcology_project
				is_gestalt = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			planet_housing_add = 1
			#job_diplomat_drone_add = 1
		}
	}
	triggered_planet_modifier = {		# Fallen Empire Housing
		potential = {
			exists = owner
			owner = {
				is_fallen_empire = yes
			}
		}
		modifier = {
			planet_housing_add = 4
		}
	}
	
	ai_weight = {	# build a couple on every arcology
		weight = 0
		
		modifier = {
			weight = 100
