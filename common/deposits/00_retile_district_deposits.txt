# Precinct District
retile_deposit_precinct = {
	is_for_colonizeable = yes
	category = deposit_cat_rare
	
	resources = {
		category = planet_deposits
		upkeep = {
			energy = 5
			#alloys = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 2
		planet_max_districts_add = -1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { NOT = { has_policy_flag = criminal_military_police } }
		}
		job_enforcer_add = 2
	}
	triggered_planet_modifier = {
		potential = {
			is_planet_class = pc_city
			exists = owner
			owner = { OR = {
				AND = {
					has_ascension_perk = ap_arcology_project
					NOT = { has_policy_flag = criminal_military_police }
				}
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			} }
		}
		modifier = {
			planet_housing_add = 1
			job_soldier_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = criminal_military_police }
		}
		job_soldier_add = 2
	}
	triggered_planet_modifier = {
		potential = {
			is_planet_class = pc_city
			exists = owner
			owner = { OR = {
				AND = {
					has_ascension_perk = ap_arcology_project
					has_policy_flag = criminal_military_police
				}
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			} }
		}
		modifier = {
			planet_housing_add = 1
			job_soldier_add = 1
		}
	}
	
	drop_weight = {
		weight = 0
	}
}
