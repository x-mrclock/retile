		exists = owner
		owner = { OR = {
			is_fallen_empire = yes
			#has_technology = retile_new_fe_tech
		} }
	}
	
	allow = {
		custom_tooltip = {
			fail_text = retile_fe_building_limit
			exists = owner
			OR = {
				owner = { is_fallen_empire = yes }
				NAND = {
					#has_building = building_dimensional_fabricator
					has_building = building_nano_forge
					has_building = building_class_4_singularity
					has_building = building_empyrean_shrine
					has_building = building_master_archive
					has_building = building_fe_xeno_zoo
					has_building = building_fe_dome
				}
			}
		}
		custom_tooltip = {
			fail_text = retile_fe_building_limit_fab
			exists = owner
			NOT = {
				owner = { any_owned_planet = {
					has_building = building_dimensional_fabricator
				} }
			}
		}
	}
	
	country_modifier = {
		country_resource_max_add = 10000
	}
	planet_modifier = {
		planet_housing_add = 4
		job_custodian_add = 4
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_custodian_add = -4
			job_fe_custodian_add = 4
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			alloys = 2000
			sr_dark_matter = 500
		}
		produces = {
			rare_crystals = 5
			exotic_gases = 5
			volatile_motes = 5
		}
		produces = {
			trigger = {
				exists = owner
				owner = { is_fallen_empire = yes }
			}
			sr_zro = 5
			sr_dark_matter = 5
			sr_living_metal = 5
		}
		upkeep = {
			energy = 500
			minerals = 500
		}
	}
}
